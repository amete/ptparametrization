# Prepare the input to the fit
# We drop everything except invabssag which we use to regress
X = df_skimmed_train.drop(['comboid','sagitta','pt','eta','phi'], axis = 1)

# Create first and second order fits and the associated predictions
predictions = [] # where we store the fit results
fits = [1,2] # orders of the fits
colors = ['black','red'] # for plotting
for count, degree in enumerate(fits):
    model = make_pipeline(PolynomialFeatures(degree), linear_model.RANSACRegressor())
    model.fit(X,df_skimmed_train.pt)
    predictions.append(model.predict(df_skimmed_test.drop(['comboid','sagitta','pt','eta','phi'], axis = 1)))
